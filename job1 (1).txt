pipeline {
    agent any
    triggers {
        pollSCM "* * * * *"
    }
    
    stages {
        stage('Get Code and install Npm Dependencies') {
            steps {
                // Get some code from a GitHub repository
                    checkout([$class: 'GitSCM', branches: [[name: '*/master']], extensions: [], userRemoteConfigs: [[url: 'https://github.com/checkmarx-ltd/cx-common-js-client.git']]])
                    sh 'cd /var/lib/jenkins/workspace/${JOB_NAME} && npm install'
            }

            post {
                // If Install Success then publish
                success {
                    sh 'cd /var/lib/jenkins/workspace/${JOB_NAME} && npm publish'
                }
            }
        }
    }
}
